# Server configuration
[server]
bind_address = 0.0.0.0
port = 8080
worker_processes = 4
timeout = 30
keepalive = 75
max_requests = 1000

[server.ssl]
enabled = true
certificate = /etc/ssl/certs/server.crt
private_key = /etc/ssl/private/server.key
protocols = [TLSv1.2, TLSv1.3]

[server.logging]
level = info
file = /var/log/server.log
max_size = 10485760
rotate = true
